;;;bind variables with a procedure
(define (outside x y)
	(define (binder a b)
		(+
		(* x a)
		(+ y b)
		)
	)
	(binder (+ x 1) (+ y 1))
)


;;; bind variables with special form let
;;; (let (<variable> <expression>)
;;;    	 (<variable> <expression>)...)
;;;	 <body>

(define (outside x y)
	(let (a (+ x 1))
	     (b (+ y 1)))
	(+ (* x a) (+ y b))
)
